# 0. 先看如何定义本地存在的二进制库文件, IMPORTED_IMPLIB 是 windows 特有属性, 用于链接

set(CUR_DIR ${CMAKE_CURRENT_SOURCE_DIR})

if(WIN32)
    add_library(add SHARED IMPORTED GLOBAL)
    set_target_properties(add PROPERTIES IMPORTED_IMPLIB ${CUR_DIR}/windows/add.lib)
    set_target_properties(add PROPERTIES IMPORTED_LOCATION ${CUR_DIR}/windows/add.dll)
elseif(LINUX)
    add_library(add SHARED IMPORTED GLOBAL)
    set_target_properties(add PROPERTIES IMPORTED_LOCATION ${CUR_DIR}/linux/libadd.so)
else()
    # 其他平台没有现成库则使用源码
    add_library(add SHARED "src/add.c")
endif()

# 1. 自定义宏 autocopy_files, 支持按需导入

# ==============================================================================
# 自动拷贝动态库
# ==============================================================================
if(NOT COMMAND autocopy_files)
    macro(autocopy_files libname target_exe)
        if(TARGET ${libname})
            get_target_property(_lib_type ${libname} TYPE)
            set(_dll_path "$<TARGET_FILE:${libname}>")
            if(_lib_type STREQUAL "SHARED_LIBRARY" AND _dll_path)
                add_custom_command(TARGET ${target_exe} POST_BUILD
                    COMMAND ${CMAKE_COMMAND} -E copy_if_different "${_dll_path}" "$<TARGET_FILE_DIR:${target_exe}>"
                    COMMAND ${CMAKE_COMMAND} -E echo "== Copying ${_dll_path}")
            endif()
        endif()
    endmacro()
endif()


# 使用
autocopy_files(add ${PROJECT_NAME})


# 2. 直接拷贝所有动态库依赖, 只对 windows 有效

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_RUNTIME_DLLS:${PROJECT_NAME}> $<TARGET_FILE_DIR:${PROJECT_NAME}> COMMAND_EXPAND_LISTS
)

# 3. 非标准拷贝任意多文件

# 可以添加一个属性 COPY_FILES, 用于指定需要拷贝的文件路径, set_target_properties 属性不支持设置多个参数, 一个属性一个参数, 
# 可以通将多个文件路径保存到变量, 再引用变量, 则属性会变成 ; 分隔的字符串

if(WIN32)

    set(VLD_COPY_FILES
        "${CUR_DIR}/bin/Win64/vld_x64.dll"
        "${CUR_DIR}/bin/Win64/dbghelp.dll"
        "${CUR_DIR}/bin/Win64/Microsoft.DTfW.DHL.manifest"
        "${CUR_DIR}/bin/Win64/vld_x64.pdb"
    )

    add_library(vld SHARED IMPORTED GLOBAL)
    set_target_properties(vld PROPERTIES
        IMPORTED_IMPLIB "${CUR_DIR}/lib/Win64/vld.lib"
        IMPORTED_LOCATION "${CUR_DIR}/bin/Win64/vld_x64.dll"
        COPY_FILES "${VLD_COPY_FILES}"
    )
endif()

# 然后在辅助宏中添加对 COPY_FILES 属性的处理

# ==============================================================================
# 自动拷贝动态库辅助宏
# ==============================================================================
if(NOT COMMAND autocopy_files)
    macro(autocopy_files libname target_exe)
        if(TARGET ${libname})

            # 处理动态库依赖
            get_target_property(_lib_type ${libname} TYPE)
            set(_dll_path "$<TARGET_FILE:${libname}>")
            if(_lib_type STREQUAL "SHARED_LIBRARY" AND _dll_path)
                add_custom_command(TARGET ${target_exe} POST_BUILD
                    COMMAND ${CMAKE_COMMAND} -E copy_if_different "${_dll_path}" "$<TARGET_FILE_DIR:${target_exe}>"
                    COMMAND ${CMAKE_COMMAND} -E echo "== Copying ${_dll_path}"
                )
            endif()

            # 处理 COPY_FILES 属性指定的文件或目录
            get_target_property(_copy_files ${libname} COPY_FILES)
            if(_copy_files)
                # 遍历每个文件（自动按 ; 拆分）
                foreach(FILE_PATH ${_copy_files})
                    add_custom_command(TARGET ${target_exe} POST_BUILD
                        COMMAND ${CMAKE_COMMAND} -E copy_if_different
                            "${FILE_PATH}"
                            "$<TARGET_FILE_DIR:${target_exe}>"
                        COMMAND ${CMAKE_COMMAND} -E echo "== Copying ${FILE_PATH}"
                    )
                endforeach()
            endif()

        endif()
    endmacro()
endif()
